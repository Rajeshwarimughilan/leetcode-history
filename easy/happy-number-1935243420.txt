class Solution {
    public boolean isHappy(int n) {
        int slow = n;
        int fast = n;

        do{
            fast = helperfindsq(helperfindsq(fast));
            slow = helperfindsq(slow);
        }while(slow != fast);

        return slow == 1;
    }

    static int helperfindsq(int x){
        int res = 0;
        while(x > 0){
                int rem = x % 10;
                res += rem * rem;;
                x = x/10;
            }
        return res;
    }
}
