class Solution {
    public boolean repeatedSubstringPattern(String s) {
        int n = s.length();
        for(int i  = 1; i <= n/2; i++){
            if(n % i == 0){
                String pattern = s.substring(0, i);
                boolean mark = true;
                for(int j = i; j < n; j += i){
                    String sub = s.substring(j, j + i);
                    if(!sub.equals(pattern)){
                        mark = false;
                    }
                }
                if(mark)return true;
            }
        }
        
        return false;

    }
}
