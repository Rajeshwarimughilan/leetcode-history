class Solution {
    public int findNumbers(int[] nums) {
        int res = 0;
        for(int i = 0; i < nums.length; i++){
            int current = nums[i];
            int count = 0;
            while(current > 0){
                int rem = current % 10;
                count++;
                current /= 10;
            }
            if(count % 2 == 0){
                res++;
            }
        }
        return res;
    }
}
