class Solution {
    public String toHex(int num) {
        if(num == 0) return "0";
        char[] hex = "0123456789abcdef".toCharArray(); 
        StringBuilder res = new StringBuilder();

        while(num != 0){
            int digit = num & 15;
            res.append(hex[digit]);
            num >>>= 4;
        }
        return res.reverse().toString();
    }
}
