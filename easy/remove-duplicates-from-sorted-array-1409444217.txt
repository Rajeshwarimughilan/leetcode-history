class Solution {
    public int removeDuplicates(int[] nums) {
        int p = nums.length;
        if(p == 0){
            return 0;
        }
        int j = 0;
        for(int i = 1; i< p; i++){
            if(nums[i] != nums[j]){
                j++;
                nums[j] = nums[i];
            }
        }
        return j+1;
    }
}
