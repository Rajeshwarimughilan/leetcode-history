class Solution {
    public int findNumbers(int[] nums) {
        int res = 0; 
        for(int i = 0; i < nums.length; i++){
            int current = nums[i];
            if(current < 0) current *= -1;
            int count = (int)(Math.log10(current)) + 1;
            
            if(count % 2 == 0){
                res++;
            }
        }
        return res;
    }
}
