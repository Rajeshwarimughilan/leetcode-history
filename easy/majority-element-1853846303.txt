class Solution {
    public int majorityElement(int[] nums) {
        HashMap<Integer, Integer> map =  new HashMap<>();
        int m = nums.length;
        for(int n: nums){
            map.put(n, map.getOrDefault(n, 0) + 1);

            if(map.get(n) > m/2) return n;
        }
        return -1;
    }
}
