class Solution {
    public boolean isSubsequence(String s, String t) {
        if(s.length() == 0) return true;
        if(t.length() == 0) return false;
        int n = s.length();
        int m = t.length();
        int j = 0;
        for(int i = 0; i < m; i++){
            if(j < n && t.charAt(i) == s.charAt(j)) j++;
            if(j == n) return true;
        }
        return false;
    }
}
