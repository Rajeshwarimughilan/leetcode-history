class Solution {
    public int maxProfit(int[] prices) {
        int n = prices.length;
        int maxprofit = Integer.MIN_VALUE;
        int minprice = prices[0];
        for(int i = 0; i  < n; i++){
            if(prices[i] < minprice) minprice = prices[i];
            maxprofit = Math.max(maxprofit, prices[i] - minprice);
        }
        return maxprofit;
    }
}
