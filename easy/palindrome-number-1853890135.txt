class Solution {
    public boolean isPalindrome(int x) {
        if(x % 10 == 0 && x != 0) return false;
        int ori = x;
        int rev = 0;
        int digit = 0;
        while(x > 0){
            digit = x % 10;
            rev = rev * 10 + digit;
            x = x/10;
        }
        if(ori == rev) return true;
        else return false;
    }
}
