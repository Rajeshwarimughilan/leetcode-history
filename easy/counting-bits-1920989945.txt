class Solution {
    public int[] countBits(int n) {
        int[] res = new int[n+1];
        for(int i = 0; i <= n; i++){
            res[i] = countsetbit(i);
        }
        return res;
    }

    public static int countsetbit(int a){
        int c = 0;
        while(a != 0){
            c++;
            a = a & (a-1);
        }
        return c;
    }
}
