class Solution {
    public int countNegatives(int[][] grid) {
        int res = 0;int n = grid.length;
        for(int i = 0; i < n; i++){
            res += helperbinarysearch(grid[i]);
        }
        return res;
    }

    public static int helperbinarysearch(int[] arr){
        int firstneg = arr.length;
        int low = 0; int high = arr.length - 1;
        while(low <= high){
            int mid = low + (high - low)/2;
            if(arr[mid] < 0){
                firstneg = mid;
                high = mid - 1;
            }else{
                low = mid + 1;
            }
        }
        return arr.length - firstneg;
    }
}
