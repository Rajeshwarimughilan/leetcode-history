class Solution {
    public int findLucky(int[] arr) {
        HashMap<Integer, Integer> map = new HashMap<>();
        int n = arr.length;
        for(int i = 0; i < arr.length; i++){
            map.put(arr[i], map.getOrDefault(arr[i], 0) + 1);
        }
        int max = -1;
        for(int i = 0; i < n; i++){
            if(arr[i] == map.get(arr[i])){
                if(max < arr[i]){
                    max = arr[i];
                }
            }
        }
        return max;
    }
}
