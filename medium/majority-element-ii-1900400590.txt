class Solution {
    public List<Integer> majorityElement(int[] nums) {
        HashMap<Integer,Integer> map = new HashMap<>();
        List<Integer> res = new ArrayList<>();

        int n = nums.length;
        for(int i  = 0; i < n; i++){
            map.put(nums[i], map.getOrDefault(nums[i], 0)+1);
        }
        int target = n/3;
        for(int key : map.keySet()){
            if(map.get(key) > target){
                res.add(key);
            }
        }
        return res;
    }
}
