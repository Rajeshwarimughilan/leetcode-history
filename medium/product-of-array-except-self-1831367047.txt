class Solution {
    public int[] productExceptSelf(int[] nums) {
        int n = nums.length;
        int[] answer = new int[n];
        int l_prod = 1;
        int r_prod = 1;

        for (int i = 0; i < n; i++){
            answer[i] = l_prod;
            l_prod *= nums[i];
        }
        
        for (int i = n-1; i >= 0; i--){
            answer[i] *= r_prod;
            r_prod *= nums[i];
        }
    return answer;    
    }
}
