/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public List<List<Integer>> zigzagLevelOrder(TreeNode root) {
        Deque<TreeNode> dq = new LinkedList<>();
        List<List<Integer>> res = new ArrayList<>();

        if(root == null) return res;
        dq.offer(root);
        boolean reverse = false;

        while(!dq.isEmpty()){
            int size = dq.size();
            ArrayList<Integer> lvl = new ArrayList<>();
            for(int i = 0; i < size; i++){
                if(!reverse){
                    //normal
                    TreeNode temp = dq.pollFirst();
                    lvl.add(temp.val);

                    if(temp.left != null) dq.addLast(temp.left);
                    if(temp.right != null) dq.addLast(temp.right);
                    }
                else{
                    //reverse
                    TreeNode temp = dq.pollLast();
                    lvl.add(temp.val);

                    if(temp.right != null) dq.addFirst(temp.right);
                    if(temp.left != null) dq.addFirst(temp.left);
                }
            }
            reverse = !reverse;
            res.add(lvl);
        }
        return res;
    }
    
}
