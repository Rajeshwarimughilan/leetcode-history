class Solution {
    public int lengthOfLongestSubstring(String s) {
       int l = 0;
       int n = s.length();
       int maxlen = 0;
       HashMap<Character, Integer> map = new HashMap<>();

       for(int r = 0; r < n; r++){
       char ch = s.charAt(r);
       if(map.containsKey(ch) && map.get(ch) >= l){
        l = map.get(ch) + 1;
        }
        
        map.put(ch, r);
        maxlen = Math.max(maxlen, r-l+1);
        
       } 
       return maxlen;
    }
}
