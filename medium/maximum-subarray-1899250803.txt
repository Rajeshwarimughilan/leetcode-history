class Solution {
    public int maxSubArray(int[] nums) {
        int val = nums[0], total = nums[0];
        int n = nums.length;
        for(int i = 1; i < n; i++){
            val = Math.max(val + nums[i], nums[i]);
            total = Math.max(total, val);
        }
        return total;
    }
}
