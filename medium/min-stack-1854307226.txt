class MinStack {
    Deque<Integer> stack;
    Deque<Integer> min;

    public MinStack() {
        min = new ArrayDeque<>();
        stack = new ArrayDeque<>();
        
    }
    
    public void push(int val) {
        stack.push(val);
        if(min.isEmpty() || val <= min.peek()) min.push(val);
        
    }
    
    public void pop() {
        
        int x = stack.pop();
        if(min.peek().equals(x)) min.pop();
    }
    
    public int top() {
        return stack.peek();
        
    }
    
    public int getMin() {
        return min.peek();
        
    }
}
