class Solution(object):
    def jump(self, nums):
        end = 0
        maxR = 0
        jump = 0
        for i in range (len(nums)-1):
            maxR = max(maxR, i+nums[i])
            if i == end:
                jump = jump + 1
                end = maxR
        return jump             
        
        '''jumps: total number of jumps you’ve made.
        end: the end of the current range (when to make a jump).
        farthest: the farthest index you can reach from the current range.'''
