class Solution(object):
    def nextPermutation(self, nums):
        n = len(nums)
        i = n-1
        while i > 0 and nums[i] <= nums[i-1]:
            i = i-1
        if i == 0:
            nums.reverse()
            return    

        pivot = i - 1
        j = n-1
        while nums[j] <= nums[pivot]:
            j = j-1

        nums[pivot], nums[j] = nums[j], nums[pivot]

        nums[i:] = nums[i:][::-1]
