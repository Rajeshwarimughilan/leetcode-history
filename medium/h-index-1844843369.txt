class Solution {
    public int hIndex(int[] citations) {
        Arrays.sort(citations);
        int n = citations.length;
        for(int i = 0; i < n/2; i++){
            int temp = citations[i];
            citations[i] = citations[n-i-1];
            citations[n-i-1] = temp;
        }
        int j = 0;
        int h = 0;
        while(j < n){
            if(citations[j] >= j+1){
                h++;
            }
            j++;
        }
        return h;
    }
}
