class Solution {
    public boolean canPartition(int[] nums) {
        int total = 0;
        for(int x : nums) total += x;
        if(total % 2 != 0) return false;
        int target = total / 2;

        boolean[] dp = new boolean[target + 1];
        dp[0] = true;//base initialization meaning 0 can be acheived wit 0 elements
        for(int n : nums){
            for(int s = target; s >= n; s--){
                dp[s] = dp[s] || dp[s - n]; 
            }
        }
        return dp[target];
        
    }
}
