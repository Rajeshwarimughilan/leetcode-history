class Solution {
    public List<List<Integer>> subsets(int[] nums) {
        List<List<Integer>> result = new ArrayList<>();
        helper(new ArrayList<>(), nums, result);
        return result;
    }

    static void helper(ArrayList<Integer> p, int[] up, List<List<Integer>> result){
        if(up.length == 0){
            result.add(new ArrayList<>(p));
            return;
        }
        int curr = up[0];
        p.add(curr);
        helper(p, Arrays.copyOfRange(up, 1, up.length), result);
        p.remove(p.size() - 1);
        helper(p, Arrays.copyOfRange(up, 1, up.length), result);
    }
}
