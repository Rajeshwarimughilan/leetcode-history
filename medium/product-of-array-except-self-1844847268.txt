class Solution {
    public int[] productExceptSelf(int[] nums) {
        int n = nums.length;
        int[] ans = new int[n];
        int l_prod = 1;
        int r_prod = 1;

        for(int i = 0;i < n; i++){
            ans[i] = l_prod;
            l_prod *= nums[i];
        }

        for(int j = n-1; j>=0; j--){
            ans[j] *= r_prod;
            r_prod *= nums[j]; 
        }
        return ans;
    }
}
