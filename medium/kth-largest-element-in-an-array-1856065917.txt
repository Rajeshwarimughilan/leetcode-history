class Solution {
    public int findKthLargest(int[] nums, int k) {
        PriorityQueue<Integer> maxheap = new PriorityQueue<>(Collections.reverseOrder());
        for(int i = 0; i < nums.length; i++){
            maxheap.add(nums[i]);
        }

        for(int j = 0; j < k-1; j++){
            maxheap.poll();
        }
        return maxheap.peek();

    }
}
