class Solution {
    public int minEatingSpeed(int[] piles, int h) {
        int left = 1;
        int right = 0;
        for(int p : piles){
            right = Math.max(right, p);
        }

        while(left < right){
            int mid = left + (right - left)/2;
            int hours = 0;
            for(int pi : piles){
                hours += (pi + mid - 1)/mid;
                //ciel func to calculate step
            }
            if(hours <= h){
                right = mid;
            }else{
                left = mid + 1;
            }
        }
        return left;
    }
}
